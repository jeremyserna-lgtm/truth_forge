**Live Demo**: [Opens BigQuery]
```sql
-- Show multi-tier truth architecture
SELECT
  tier,
  confidence_level,
  COUNT(*) as data_points
FROM (
  SELECT
    CASE
      WHEN source = 'sms' AND verified = true THEN 'Tier 1: Ground Truth'
      WHEN source = 'ai' AND v_cmp IS NOT NULL THEN 'Tier 2: Derived Truth'
      ELSE 'Tier 3: Hypothesis'
    END as tier,
    CASE
      WHEN source = 'sms' THEN 1.0
      WHEN v_cmp >= 0.9 THEN 0.95
      WHEN v_cmp >= 0.7 THEN 0.85
      ELSE 0.70
    END as confidence_level
  FROM source_data.ai_conversations_enriched
)
GROUP BY tier, confidence_level
```
